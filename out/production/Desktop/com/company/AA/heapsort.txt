// Algorithm of heapify
heapify(array, n, index) {
    size = array.size()
largest = index
    left = 2 * index + 1
    right = 2 * index + 2
    if left < n && arr[left] > arr[largest]
        largest = left
    if right < n && arr[right] > arr[largest]
        largest = right
      if largest != index {
        swap arr[index], arr[largest]
        heapify(arr, n, largest)
    }
}
